import os, random, sys;

$dict stats = !dict;

for $int a = 0; a < 1000; a += 1 {
    $int i = !random.randint 0, 100;
    $str p = i % 10 * 10;
    if p in stats { $int stats[p] += 1; }
    else { $int stats[p] = 1; }
}

$list b = !sorted stats;
for $int a = 0; a < !len b; a += 1 {
    $float p = stats[b[a]] / 1000 * 100;
    $str r = !round p, 3;
    !print !str b; + ': ' + r + '%';
}
