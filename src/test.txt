import random, sys;

class Main {
    fun __init__ self {
        // Stuff.
        $str a = 10;
        // Making variables here.
        $self.get_input a = 'Enter '
        'file '
        'name '
        'here: ';
        !self.importFile a;
    }

    fun get_input self prompt {
        if sys.version_info[0] < 3 {
            return !raw_input prompt;
        } else {
            return !input prompt;
        }
    }

    fun findFile self fileName {
        $var FileNotFound = ? sys.version_info[0] < 3 :: IOError | FileNotFoundError;
        try {
            $open f = fileName;
            !f.close;
            return true;
        } catch FileNotFound {
            return false;
        }
    }

    fun importFile self fileName {
        if !self.findFile fileName {
            !print 'File exists.';
        } else {
            !print "File doesn't exist.";
        }
    }

    fun roundNumber self num places=0 {
        return !round num, places;
    }
}
